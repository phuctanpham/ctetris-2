\section{Đặc tả Yêu cầu Nghiệp vụ (SRS): Module gameConsole}

\subsection{Tổng quan Nghiệp vụ (Business Overview)}
Phân hệ \textbf{gameConsole} đóng vai trò là Trung tâm điều phối (Hub Screen) của ứng dụng \textit{cTetris}. Module này quản lý trạng thái của hệ thống ngoài luồng chơi chính, cung cấp giao diện để người dùng khởi động ván đấu, tra cứu hướng dẫn, xem bảng xếp hạng, thiết lập cấu hình cá nhân hóa và quản lý việc thoát ứng dụng an toàn trên đa nền tảng.

    \textbf{3. Kiểm tra} & Người chơi dùng phím TAB để chuyển viền vàng xuống nút BOARD. Bảng xếp hạng hiện ra với danh sách 30 người chơi hàng đầu. \\

    \hline
    \textbf{5. Kết thúc} & Nếu người chơi muốn nghỉ, họ chọn QUIT. Trên trình duyệt, một nút RELOAD lớn xuất hiện để họ có thể dễ dàng bắt đầu lại phiên mới sau này. \\

\subsection{Yêu cầu Phiên bản V1: Nền tảng Hub Screen \& Lightbox UI}
\begin{itemize}[leftmargin=*]
    \item \textbf{Mục tiêu cốt lõi:} Xây dựng bộ khung UI tiêu chuẩn, hệ thống điều hướng đa kênh (chuột, phím, con lăn) và quản lý Pop-up (Lightbox) bằng mảng dữ liệu tĩnh.
    \item \textbf{Yêu cầu Chức năng:}
    \begin{itemize}[noitemsep]
        \item Hệ thống Pop-up Lightbox: Phủ mờ nền 70\%, tích hợp Scrollbar hỗ trợ kéo thả và tự động cuộn.
        \item Điều hướng đa năng: Hỗ trợ WASD, Mũi tên, TAB cho việc chuyển đổi Focus; Enter/Space để chọn.
        \item Xử lý văn bản: Thuật toán Word Wrapping tự động ngắt dòng tại 28 ký tự để đảm bảo hiển thị đẹp trong popup.
    \end{itemize}
    \item \textbf{Yêu cầu Phi Chức năng:} Đảm bảo thời gian phản hồi phím < 50ms. Tích hợp ký hiệu Copyright (©) UTF-8 trên thanh tiêu đề OS.
    \item \textbf{Yêu cầu Kỹ thuật \& Tiêu chí thành công:}
    \begin{itemize}[noitemsep]
        \item Kỹ thuật: Sử dụng cấu trúc \texttt{gameConsole\_layout.h}. Chế độ \texttt{BUILD\_STANDALONE} cho kiểm thử độc lập.
        \item \textbf{Tiêu chí thành công:} Chuyển đổi trạng thái Focus mượt mà; Popup không bị lỗi tràn văn bản; Scrollbar hoạt động chính xác theo tọa độ chuột.
    \end{itemize}
\end{itemize}

\subsection{Yêu cầu Phiên bản V2: Cấu hình cá nhân hóa \& JSON Data}
\begin{itemize}[leftmargin=*]
    \item \textbf{Mục tiêu cốt lõi:} Tăng cường trải nghiệm bằng hình nền động (Dynamic Background), thay thế dữ liệu cứng bằng File JSON nội bộ và bổ sung tính năng cá nhân hóa Settings.
    \item \textbf{Yêu cầu Chức năng:}
    \begin{itemize}[noitemsep]
        \item Hình nền động: Hỗ trợ tải ảnh nền từ file vật lý với thuật toán co dãn bảo toàn tỷ lệ 9:16.
        \item Màn hình Settings: Cho phép điều chỉnh âm lượng (Slider) và lựa chọn màu sắc khối Tetromino chủ đạo.
        \item Dữ liệu cục bộ: Đọc/Ghi thông tin Bảng xếp hạng và thiết lập người dùng xuống tệp JSON.
    \end{itemize}
    \item \textbf{Yêu cầu Kỹ thuật \& Tiêu chí thành công:}
    \begin{itemize}[noitemsep]
        \item Kỹ thuật: Tích hợp thư viện parse JSON. Cơ chế đồng bộ thiết lập (Settings) sang module \texttt{gameCore}.
        \item \textbf{Tiêu chí thành công:} Thiết lập của người dùng được lưu lại và áp dụng chính xác cho ván đấu tiếp theo.
    \end{itemize}
\end{itemize}

\subsection{Yêu cầu Phiên bản V3: Tích hợp Đám mây \& Bảng xếp hạng Động}
\begin{itemize}[leftmargin=*]
    \item \textbf{Mục tiêu cốt lõi:} Số hóa tính cạnh tranh bằng cách đồng bộ Bảng xếp hạng trực tuyến qua API MongoDB Atlas, hỗ trợ bộ lọc và sắp xếp (Sorting).
    \item \textbf{Yêu cầu Chức năng:}
    \begin{itemize}[noitemsep]
        \item Đồng bộ trực tuyến: Gọi HTTP Request lấy dữ liệu Leaderboard thời gian thực.
        \item Phân loại dữ liệu: Chức năng sắp xếp danh sách theo "Điểm cao" hoặc "Thời gian gần nhất".
        \item Trạng thái mạng: Hiển thị hiệu ứng Loading khi đang truy vấn dữ liệu từ Server.
    \end{itemize}
    \item \textbf{Yêu cầu Phi Chức năng:} Xử lý bất đồng bộ (Asynchronous) để không làm treo giao diện chính khi mạng yếu. Timeout tối đa 2 giây.
    \item \textbf{Yêu cầu Kỹ thuật \& Tiêu chí thành công:}
    \begin{itemize}[noitemsep]
        \item Kỹ thuật: Client Network sử dụng \texttt{curl} hoặc \texttt{emscripten\_fetch}. Thuật toán sắp xếp mảng BoardEntry tại Client.
        \item \textbf{Tiêu chí thành công:} Bảng xếp hạng hiển thị đúng dữ liệu trên Cloud; Ứng dụng xử lý lỗi mạng (Offline mode) nhịp nhàng.
    \end{itemize}
\end{itemize}
