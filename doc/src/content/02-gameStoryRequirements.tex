% ---- Yêu cầu chức năng V2 ----
\subsubsection{Yêu cầu Chức năng V2}
\begin{itemize}[leftmargin=*]
    \item \textbf{Dual-mode entry:} \texttt{storyId=0} → intro + sync; \texttt{storyId>0} → dialogue trực tiếp (từ Console Play/Stories popup). Backward-compatible với \texttt{main.cpp} V1 call.
    \item \textbf{Gist SHA-diff sync:} so sánh \texttt{shared\_meta.sha} vs manifest; chỉ fetch và apply chapters có SHA mới. Offline → \texttt{SYNC\_OFFLINE}, tiếp tục dùng local data.
    \item \textbf{applyChapterJson():} DELETE stories no longer in JSON → UPSERT \texttt{shared\_data} + \texttt{shared\_dialogues} + \texttt{shared\_choices} trong một transaction. Không full-replace để bảo toàn user progress.
    \item \textbf{Dialogue engine:} load nodes bulk → \texttt{dialRunLoop()} state machine; choices loaded on-demand per node. Terminal: \texttt{nextNodeId=0} → exit.
    \item \textbf{Skip button:} visible ONLY trong PHASE\_LOGO (sync đã hoàn thành); ẩn trong phase sync để tránh user bỏ qua DB write.
    \item \textbf{Progress bar V2:} \texttt{fill = max(syncProgress, timeProgress)}; status line hiển thị chapter đang sync / "Offline" / "Up to date".
    \item \textbf{Post-sync condition check:} return 0=Console, +N=replay, $-$N=prompt unlock.
\end{itemize}

\subsubsection{Yêu cầu Phi Chức năng V2}
\begin{itemize}[noitemsep, leftmargin=*]
    \item \textbf{HTTP:} native = libcurl blocking (startup sync acceptable); WASM = \texttt{emscripten\_fetch} sync mode qua Asyncify (\texttt{-sASYNCIFY=1}).
    \item \textbf{SQLite:} gameStory chỉ \emph{đọc} \texttt{shared\_dialogues/choices}; \emph{ghi} chỉ vào \texttt{shared\_meta} (SHA tracking) và \texttt{idUser\_Stories} (story selection). Không ghi \texttt{idUser\_Records}.
    \item \textbf{WASM persistence:} \texttt{EM\_ASM(\{Module.FS.syncfs(false,\ldots)\})} sau mỗi chapter apply và sau story selection update.
    \item \textbf{Offline degradation:} \texttt{MANIFEST\_GIST\_URL} rỗng hoặc fetch fail → \texttt{SYNC\_OFFLINE} → tiếp tục bình thường với local SQLite data.
\end{itemize}

\subsection{Yêu cầu Phiên bản V3: Tích hợp API \& Lưu trữ Đám mây}
\begin{itemize}[leftmargin=*]
    \item \textbf{Mục tiêu cốt lõi:} Tối ưu dung lượng bằng cách tải tài nguyên Media động từ Cloud dựa trên tốc độ mạng thực tế.
    \item \textbf{Đặc tả Giao diện \& Hiệu ứng:} Trạng thái Download (KB/s, \%); Hiệu ứng Logo lặp (Pulse) báo hiệu đang tải.
    \item \textbf{Luồng tương tác sử dụng (userflow):}
    \begin{center}
    \begin{tikzpicture}[node distance=1cm, every node/.style={font=\scriptsize}, start/.style={circle, draw, fill=green!20}, end/.style={circle, draw, fill=red!20, double}, process/.style={rectangle, draw, fill=orange!10}, decision/.style={diamond, draw, fill=blue!5, aspect=2}]
        \node (s) [start] {Open};
        \node (d1) [decision, below=0.6cm of s] {Online?};
        \node (p1) [process, below=0.6cm of d1] {Tải Assets};
        \node (p2) [process, below=0.6cm of p1] {Chạy Story (V2)};
        \node (e) [end, right=1.5cm of d1] {End (V1)};
        \draw [->] (s) -- (d1);
        \draw [->] (d1) -- node[left] {Yes} (p1); \draw [->] (d1) -- node[above] {No} (e);
        \draw [->] (p1) -- (p2); \draw [->] (p2.east) -| (e.south);
    \end{tikzpicture}
    \end{center}
    \item \textbf{Sơ đồ tình huống sử dụng (usecase):}
    \begin{center}
    \begin{tikzpicture}[node distance=1cm, every node/.style={font=\scriptsize}, actor/.style={circle, draw, fill=gray!10}, usecase/.style={ellipse, draw, fill=white, minimum width=2.5cm}]
        \draw (0,0) rectangle (5,-3); \node at (2.5,-0.3) {\textbf{System V3}};
        \node (sys) [actor] at (6.5, -1.5) {Cloud API};
        \node (u1) [usecase] at (2.5,-1) {Kiểm tra mạng};
        \node (u2) [usecase] at (2.5,-2) {Tải Media};
        \draw [->] (sys) -- (u2);
    \end{tikzpicture}
    \end{center}
    \item \textbf{Yêu cầu Chức năng:} Đồng bộ Loading Bar với tiến độ download; Tự động Skip nếu mất mạng; Xóa media nặng khỏi file build.
    \item \textbf{Yêu cầu Phi Chức năng:} Cơ chế Timeout 15s; Bảo mật tài nguyên; Chống treo luồng chính (Main Thread).
    \item \textbf{Yêu cầu Kỹ thuật:} \texttt{emscripten\_fetch} (Web) hoặc Networking (Native); Quản lý Cache tài nguyên động.
    \item \textbf{Tiêu chí thành công (user acceptance criteria):} Giảm 50\% dung lượng file .exe; Cốt truyện chỉ chạy sau khi tải xong 100\%.
\end{itemize}
