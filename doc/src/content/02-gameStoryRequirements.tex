\section{Đặc tả Yêu cầu Nghiệp vụ (SRS): Module gameStory}

\subsection{Tổng quan Nghiệp vụ (Business Overview)}
Phân hệ \textbf{gameStory} đóng vai trò là màn hình khởi động và giới thiệu của ứng dụng cTetris. Module chịu trách nhiệm thiết lập nhận diện thương hiệu, tải tài nguyên hệ thống, và dẫn dắt người chơi vào không gian trò chơi thông qua cốt truyện tương tác trước khi chuyển sang Menu chính (\texttt{gameConsole}).

\subsection{Yêu cầu Phiên bản V1: Khởi tạo \& Nhận diện Thương hiệu}
\begin{itemize}[leftmargin=*]
    \item \textbf{Mục tiêu cốt lõi:} Thiết lập nền tảng hoạt động độc lập và nhận diện thương hiệu cTetris thông qua hiệu ứng thị giác 8 giây tối ưu.
    \item \textbf{Đặc tả Giao diện \& Hiệu ứng:}
    \begin{itemize}[noitemsep]
        \item Layout dọc tỷ lệ 9:16; Logo cTetris (3$\times$3 ô vuông) Fade-in chính giữa.
        \item Hiệu ứng văn bản mỏng (Thinning) cho tên game; Loading bar xanh lá (180$\times$12 px).
    \end{itemize}
    \item \textbf{Luồng tương tác sử dụng (userflow):}
    \begin{center}
    \begin{tikzpicture}[node distance=1cm, every node/.style={font=\scriptsize}, start/.style={circle, draw, fill=green!20, minimum size=0.6cm}, end/.style={circle, draw, fill=red!20, double, minimum size=0.6cm}, process/.style={rectangle, draw, fill=orange!10, minimum width=2cm}, decision/.style={diamond, draw, fill=blue!5, aspect=2}]
        \node (s) [start] {Start};
        \node (p1) [process, below=0.6cm of s] {Init Clock (T=0)};
        \node (p2) [process, below=0.6cm of p1] {Update UI/Bar};
        \node (d1) [decision, below=0.6cm of p2] {T $\geq$ 8s?};
        \node (e) [end, right=1.5cm of d1] {End};
        \draw [->] (s) -- (p1); \draw [->] (p1) -- (p2); \draw [->] (p2) -- (d1);
        \draw [->] (d1) -- node[above] {Yes} (e); \draw [->] (d1.west) -- ++(-0.5,0) |- (p2.west);
    \end{tikzpicture}
    \end{center}
    \item \textbf{Sơ đồ tình huống sử dụng (usecase):}
    \begin{center}
    \begin{tikzpicture}[node distance=1cm, every node/.style={font=\scriptsize}, actor/.style={circle, draw, fill=gray!10, minimum size=0.5cm}, usecase/.style={ellipse, draw, fill=white, minimum width=2cm}]
        \draw (0,0) rectangle (4,-2.5); \node at (2,-0.3) {\textbf{System V1}};
        \node (p) [actor] at (-1.5, -1.25) {Player};
        \node (u1) [usecase] at (2,-1) {Xem Intro};
        \node (u2) [usecase] at (2,-1.8) {Theo dõi Bar};
        \draw [->] (p) -- (u1); \draw [->] (p) -- (u2);
    \end{tikzpicture}
    \end{center}
    \item \textbf{Yêu cầu Chức năng:} Tự động chạy Intro; Lấp đầy Loading Bar tuyến tính; Chuyển sang Menu sau 8 giây.
    \item \textbf{Yêu cầu Phi Chức năng:} Tương thích đa nền tảng (SDL3); Hỗ trợ ký tự Copyright (©) UTF-8.
    \item \textbf{Yêu cầu Kỹ thuật:} Sử dụng \texttt{nanosvg}; Cơ chế Lazy Init Texture tối ưu tài nguyên.
    \item \textbf{Tiêu chí thành công (user acceptance criteria):} Hệ thống tự động chuyển Menu đúng 8.0s; Không vỡ đồ họa SVG.
\end{itemize}

\subsection{Yêu cầu Phiên bản V2: Cốt truyện Tương tác}
\begin{itemize}[leftmargin=*]

    \item \textbf{Đặc tả Giao diện \& Hiệu ứng:} Dialogue Box mờ nửa dưới; Hiệu ứng chữ máy đánh chữ; BGM đồng bộ cốt truyện.
    \item \textbf{Luồng tương tác sử dụng (userflow):}

    \item \textbf{Tiêu chí thành công (user acceptance criteria):} Người chơi hoàn thành đủ các nhánh truyện; Skip không gây lỗi logic.
\end{itemize}
