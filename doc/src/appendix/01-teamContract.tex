\section{Quy Tắc Làm Việc}
\begin{enumerate}[itemsep=0pt, parsep=0pt]
    \item \textbf{phucpt:} Phản hồi slack và cập nhật trello thường xuyên để mọi người nắm bắt tiến độ lẫn nhau.
    \item \textbf{vint:} Tuân thủ deadline đã cam kết. Nếu gặp sự cố phải thông báo ngay cho các thành viên khác hỗ trợ.
    \item \textbf{truongtm:} Trong giai đoạn cao điểm, các thành viên phải duy trì trạng thái "Online" trên Slack trong khung giờ quy định. Không tự ý thay đổi cấu trúc tệp .tex mà không có sự đồng ý của người quản lý trực tiếp.
    \item \textbf{thaidg:} Mỗi thành viên phải chịu trách nhiệm với phần việc của mình và không làm ảnh hưởng tiến độ chung của nhóm.
    \item \textbf{kiennt:} Làm việc trong Trello, không tự ý can thiệp nhánh Git của người khác, và mọi thay đổi mã nguồn đều bắt buộc phải thông qua Pull Request để review.
\end{enumerate}

\section{Giải Quyết Xung Đột}
\begin{enumerate}[itemsep=0pt, parsep=0pt]
    \item \textbf{phucpt:} Tài liệu nào do ai quản lý, người đó đưa ra quyết định cuối cùng.
    \item \textbf{vint:} Quyết định theo hình thức bỏ phiếu số đông.
    \item \textbf{truongtm:} Có quyền yêu cầu rà soát lại bảng phân công hoặc họp nhanh qua Google Meet nếu quá tải hoặc bất đồng quan điểm.
    \item \textbf{thaidg:} Mọi mâu thuẫn sẽ được thảo luận công khai và quyết định theo đa số, các thành viên có trách nhiệm tôn trọng và thực hiện.
    \item \textbf{kiennt:} Nếu hai bên không thể tự thống nhất bằng logic, quyết định cuối cùng sẽ do Trưởng nhóm phán quyết dựa trên biểu quyết hoặc tài liệu thiết kế gốc.
\end{enumerate}
