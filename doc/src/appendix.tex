\newcommand{\InputAppendixFile}[1]{%
	\IfFileExists{src/appendix/#1}{\input{src/appendix/#1}}{%
		\IfFileExists{appendix/#1}{\input{appendix/#1}}{\input{../appendix/#1}}%
	}%
} 

\makeatletter
% FIX: Safely capture the exact hyperlink anchor (\@currentHref) and write it to the .toa file
\newcommand{\IncludeAppendixPDF}[3]{%
    \clearpage
    \phantomsection
    \pdfbookmark[1]{#2}{#3}% Adds bookmark to the PDF Viewer Pane at level 1
    % Writes to the printed Phụ lục TOC (no page number) with the perfect anchor target
    \protected@write\@auxout{}{\string\@writefile{toa}{\string\contentsline{section}{#2}{}{\@currentHref}}}%
    \IfFileExists{src/appendix/#1}{%
        \includepdf[pages=-]{src/appendix/#1}%
    }{%
        \IfFileExists{appendix/#1}{%
            \includepdf[pages=-]{appendix/#1}%
        }{%
            \includepdf[pages=-]{../appendix/#1}%
        }%
    }%
}

% FIX: Apply the exact same robust linking to .tex files so they appear in the .toa file
\providecommand{\InputAppendixUnit}[2]{%
	\clearpage
	\phantomsection
	\pdfbookmark[1]{#2}{app:#1}%
	\protected@write\@auxout{}{\string\@writefile{toa}{\string\contentsline{section}{#2}{}{\@currentHref}}}%
	\InputAppendixFile{#1}%
}
\makeatother
