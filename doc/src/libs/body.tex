\newcommand{\LinearSeq}[1]{%

  \foreach \s [count=\i] in \steps {
    \node[sdsBox=pillarBlue!60] (n\i) at ({(\i-1)*3.8cm},0) {\s};
  }
  \foreach \i [count=\j from 2] in {1,...,4} {
    \draw[sdsArrow=pillarBlue] (n\i)--(n\j);
  }
  \end{tikzpicture}\end{center}
}
