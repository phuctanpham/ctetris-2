% Group Heading
\newcommand{\ContentGroup}[2]{%
  \clearpage
  \stepcounter{contentgroup}%
  \phantomsection
  \addcontentsline{toc}{part}{\textbf{#1.\ \MakeUppercase{#2}}}%
  \vspace*{0.4cm}%
  {\centering\large\bfseries #1.\ \MakeUppercase{#2}\par}% 
  \vspace{0.4cm}%
}

\newcommand{\InputContentFile}[1]{%
  % Dynamically apply the current filename to the footer. 
  % We use the built-in "LastPage" label (from the lastpage package) because manual labels at the end of the document can sometimes fail to resolve.
  \ApplyStandardFooter{#1}{LastPage}%
  \IfFileExists{src/content/#1}{\input{src/content/#1}}{\input{content/#1}}%
}

% FIX 2: Smooth out the bookmark hierarchy to prevent jump warnings.
\newcommand{\InputGroupedContentFile}[1]{%
  {%
    \let\section\subsection
    \let\subsection\subsubsection
    \let\subsubsection\paragraph
    %
    \let\origaddcontentsline\addcontentsline
    \renewcommand{\addcontentsline}[3]{%
      \def\tempa{##2}%
      \def\tempc{subsection}%
      \def\tempd{subsubsection}%
      \def\tempe{paragraph}%
      % Map the original commands to sequential outline levels
      \ifx\tempa\tempc
        \origaddcontentsline{##1}{section}{##3}%
      \else\ifx\tempa\tempd
        \origaddcontentsline{##1}{subsection}{##3}%
      \else\ifx\tempa\tempe
        \origaddcontentsline{##1}{subsubsection}{##3}%
      \else
        \origaddcontentsline{##1}{##2}{##3}%
      \fi\fi\fi
    }%
    \InputContentFile{#1}%
  }%
}
