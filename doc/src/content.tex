% Fix 3 — Group heading (Ngắt trang + In hoa tiêu đề)
\newcommand{\ContentGroup}[2]{%
  \clearpage
  \setcounter{section}{0}%
  \phantomsection
  % Tự động viết hoa tiêu đề trong Mục lục (TOC)
  \addcontentsline{toc}{section}{\textbf{#1.\ \MakeUppercase{#2}}}% 
  \vspace*{0.4cm}%
  % Tự động viết hoa tiêu đề hiển thị trong văn bản
  {\centering\large\bfseries #1.\ \MakeUppercase{#2}\par}% 
  \vspace{0.4cm}%
}
